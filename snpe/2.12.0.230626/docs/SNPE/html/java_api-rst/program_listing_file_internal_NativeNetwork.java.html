    private static final HashMap<Runtime, Boolean> mRuntimeAvailability = new HashMap<>();

    public synchronized static boolean isRuntimeAvailable(final Runtime runtime, final RuntimeCheckOption runtimeCheckOption) {
      try {
            if (!mRuntimeAvailability.containsKey(runtime)) {
                mRuntimeAvailability.put(runtime, nativeIsRuntimeAvailable(runtime.ordinal, runtimeCheckOption.ordinal));
            }
        } catch (IllegalStateException e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }

        return mRuntimeAvailability.get(runtime);
    }

    public NativeNetwork(final Runtime runtime, final NativeBufferOutputStream modelBuffer,
                         final Set<String> outputLayerNames, final boolean isDebugEnabled,
                         final PerformanceProfile performanceProfile,
                         final ExecutionPriorityHint executionPriorityHint,
                         boolean isCpuFallbackEnabled,
                         boolean unsignedPD,
                         boolean useUserSuppliedBuffers,
                         boolean isInitCacheEnabled,
                         boolean isDlCacheAvailable,
                         boolean cpuFxpMode,
                         final String storageDirectory,
                         final String dspCachePath,
                         final Map<String, int[]> inputDimensions) {
        final JniInputBundle initBundle = createInitBundle(runtime,
                outputLayerNames, isDebugEnabled, performanceProfile,
                executionPriorityHint, isCpuFallbackEnabled, unsignedPD,
                useUserSuppliedBuffers, isInitCacheEnabled, isDlCacheAvailable,
                cpuFxpMode, storageDirectory, dspCachePath, inputDimensions);
        final ByteBuffer bundleBuffer = initBundle.marshall();

        final long start = SystemClock.elapsedRealtime();
        try {
            mNativeHandle = nativeInitFromBuffer(modelBuffer.getNativeAddress(),
                    bundleBuffer, bundleBuffer.capacity());
        } catch (IllegalStateException e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }

        final long end = SystemClock.elapsedRealtime();
        if (isDebugEnabled) {
            Log.d(SNPE.LOG_TAG, "Network initialization took " + (end - start) + " ms");
        }

        init(runtime, outputLayerNames, isDebugEnabled);

    }

    private void init(Runtime runtime, Set<String> outputLayerNames, boolean isDebugEnabled) {
        final JniOutputBundle outputBundle;
        try {
            outputBundle = nativeGetModelMetadata(mNativeHandle);
        } catch (IllegalStateException e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }
        mModelVersion = outputBundle.readString();
        mOutputLayerNames = readOutputLayerNames(outputBundle);
        mInputTensors = new TensorAttributesMap(outputBundle);
        mOutputTensors = new TensorAttributesMap(outputBundle);
        mTargetRuntime = runtime;
        mIsDebugEnabled = isDebugEnabled;


        assertSelectedOutputLayer(mOutputLayerNames, outputLayerNames);
    }

    public NativeNetwork(final Runtime runtime, final File modelFile,
                         final Set<String> outputLayerNames, final boolean isDebugEnabled,
                         final PerformanceProfile performanceProfile,
                         final ExecutionPriorityHint executionPriorityHint,
                         boolean isCpuFallbackEnabled, boolean unsignedPD,
                         boolean useUserSuppliedBuffers,
                         boolean isInitCacheEnabled,
                         boolean isDlCacheAvailable,
                         boolean cpuFxpMode,
                         final String storageDirectory, final String dspCachePath,
                         final Map<String, int[]> inputDimensions) {
        final JniInputBundle initBundle = createInitBundle(runtime,
                outputLayerNames, isDebugEnabled, performanceProfile,
                executionPriorityHint, isCpuFallbackEnabled, unsignedPD,
                useUserSuppliedBuffers, isInitCacheEnabled, isDlCacheAvailable,
                cpuFxpMode, storageDirectory, dspCachePath, inputDimensions);
        final ByteBuffer bundleBuffer = initBundle.marshall();

        final long start = SystemClock.elapsedRealtime();
        try {
            mNativeHandle = nativeInitFromFile(modelFile.getAbsolutePath(),
                    bundleBuffer, bundleBuffer.capacity());
        } catch (IllegalStateException e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }
        final long end = SystemClock.elapsedRealtime();

        if (isDebugEnabled) {
            Log.d(SNPE.LOG_TAG, "Network initialization took " + (end - start) + " ms");
        }

        init(runtime, outputLayerNames, isDebugEnabled);

    }

    @Override
    public Map<String, FloatTensor> execute(final Map<String, FloatTensor> inputs) {
        if (inputs == null || inputs.isEmpty()) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_INPUT_TENSOR_NULL_EMPTY);
        }

        final Map<String, NativeFloatTensor> nativeTensorsMap = new HashMap<>();
        for (String name : inputs.keySet()) {
            final Tensor tensor = inputs.get(name);
            assertInputTensor(name, tensor);
            nativeTensorsMap.put(name, NativeFloatTensor.class.cast(tensor));
        }

        final JniInputBundle bundle = new JniInputBundle();
        final List<ByteBuffer> buffers = writeInputTensors(bundle, nativeTensorsMap);

        final ByteBuffer bundleBuffer = bundle.marshall();
        final JniOutputBundle outputBundle;
        final long start = SystemClock.elapsedRealtime();
        try {
            outputBundle = nativeForward(mNativeHandle, bundleBuffer,
                    bundleBuffer.capacity(), buffers.toArray(new ByteBuffer[buffers.size()]));
        } catch (IllegalStateException e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }
        final long end = SystemClock.elapsedRealtime();

        if (mIsDebugEnabled) {
            Log.d(SNPE.LOG_TAG, "Network forward took " + (end - start) + " ms");
        }

        return readOutputTensors(outputBundle);
    }

    @Override
    public boolean execute(Map<String, ? extends UserBufferTensor> inputs,
                           Map<String, ? extends UserBufferTensor> outputs) {
        if (inputs == null || inputs.isEmpty()) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_INPUT_TENSOR_NULL_EMPTY);
        }

        if (outputs == null || outputs.isEmpty()) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_OUTPUT_TENSOR_NULL_EMPTY);
        }

        for (String name : inputs.keySet()) {
            assertInputTensor(name, inputs.get(name));
        }

        for (String name : outputs.keySet()) {
            assertOutputTensor(name, outputs.get(name));
        }

        final JniInputBundle bundle = new JniInputBundle();
        writeUserBufferTensors(bundle, inputs);
        writeUserBufferTensors(bundle, outputs);

        final ByteBuffer bundleBuffer = bundle.marshall();

        final long start = SystemClock.elapsedRealtime();
        final JniOutputBundle outputBundle;
        try {
            outputBundle = nativeUserBufferForward(mNativeHandle, bundleBuffer, bundleBuffer.capacity());
        } catch (Exception e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }

        final long end = SystemClock.elapsedRealtime();

        if (mIsDebugEnabled) {
            Log.d(SNPE.LOG_TAG, "Network forward took " + (end - start) + " ms");
        }

        return readOutputUserBufferTensors(outputBundle, outputs);
    }

    @Override
    public FloatTensor createFloatTensor(final int... shape) {
        if (shape == null || shape.length == 0) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_TENSOR_SHAPE_NULL_EMPTY);
        }

        for (int dim : shape) {
            if (dim < 1) {
                throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_SHAPE_DIM);
            }
        }
        return new NativeFloatTensor(shape);
    }

    @Override
    public TF8UserBufferTensor createTF8UserBufferTensor(final int size, final int[] strides,
                                                         int stepExactly0, float quantizedStepSize,
                                                         ByteBuffer userBuffer) {
        if (size <= 0) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_BUFFER_SIZE);
        }

        for (int stride : strides) {
            if (stride < 1) {
                throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_STRIDE);
            }
        }

        if (stepExactly0 < TF8UserBufferTensor.TF8_MIN_VAL || stepExactly0 > TF8UserBufferTensor.TF8_MAX_VAL) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_STEP_ZERO);
        }

        if (quantizedStepSize <= 0.0) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_STEP_VALUE);
        }

        if (userBuffer == null) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_BUFFER_NULL);
        }

        return new NativeTF8UserBufferTensor(size, strides, stepExactly0, quantizedStepSize, userBuffer);
    }

            return mOutputTensors.get(name);
        } else {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_TENSOR_NAME, name);
        }
    }

    @Override
    public synchronized void release() {
        if (mNativeHandle == 0) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_NETWORK_RELEASE);
        } else {
            doRelease();
        }
    }

    @Override
    protected void finalize() throws Throwable {
        super.finalize();
        doRelease();
    }

    // Private =====================================================================================

    private void doRelease() {
        if (mNativeHandle != 0) {
            try {
                nativeRelease(mNativeHandle);
            } catch (IllegalStateException e) {
                throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
            }

            mNativeHandle = 0;
        }
    }

    private static Set<String> readOutputLayerNames(final JniOutputBundle modelMetadata) {
        final Set<String> outputNames = new HashSet<>();
        final int layersCount = modelMetadata.readInt();
        for (int i = 0; i < layersCount; i++) {
            outputNames.add(modelMetadata.readString());
        }
        return outputNames;
    }

    private static void assertSelectedOutputLayer(final Set<String> outputLayerNames,
                                                  final Set<String> selectedLayerNames) {
        for (String selectedLayer : selectedLayerNames) {
            if (!outputLayerNames.contains(selectedLayer)) {
                throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_SELECTED_LAYER, selectedLayer);
            }
        }
    }

    private static JniInputBundle createInitBundle(final Runtime runtime,
                                                   final Set<String> layerNames,
                                                   final boolean isDebugEnabled,
                                                   final PerformanceProfile performanceProfile,
                                                   final ExecutionPriorityHint executionPriorityHint,
                                                   boolean isCpuFallbackEnabled,
                                                   boolean unsignedPD,
                                                   boolean useUserSuppliedBuffers,
                                                   boolean isInitCacheEnabled,
                                                   boolean isDlCacheAvailable,
                                                   boolean cpuFxpMode,
                                                   final String storageDirectory,
                                                   final String dspCachePath,
                                                   final Map<String, int[]> inputDimensions) {
        final JniInputBundle bundle = new JniInputBundle();
        bundle.writeInt(runtime.ordinal);
        bundle.writeInt(performanceProfile.ordinal);
        bundle.writeInt(executionPriorityHint.ordinal);
        bundle.writeStrings(layerNames);
        bundle.writeBoolean(isDebugEnabled);
        bundle.writeBoolean(isCpuFallbackEnabled);
        bundle.writeBoolean(unsignedPD);
        bundle.writeBoolean(useUserSuppliedBuffers);
        bundle.writeBoolean(isInitCacheEnabled);
        bundle.writeBoolean(isDlCacheAvailable);
        bundle.writeBoolean(cpuFxpMode);
        bundle.writeString(storageDirectory);
        bundle.writeString(dspCachePath);
        bundle.writeInt(inputDimensions.size());
        for (Map.Entry<String, int[]> pair : inputDimensions.entrySet()) {
            bundle.writeString(pair.getKey());
            bundle.writeInts(pair.getValue());
        }
        return bundle;
    }

    private Map<String, FloatTensor> readOutputTensors(final JniOutputBundle outputBundle) {
        final Map<String, FloatTensor> outputsMap = new HashMap<>();
        final int outputCount = outputBundle.readInt();
        for (int i = 0; i < outputCount; i++) {
            Pair<String, FloatTensor> tuple = readTensor(outputBundle);
            outputsMap.put(tuple.first, tuple.second);
        }
        return outputsMap;
    }

    private Pair<String, FloatTensor> readTensor(final JniOutputBundle outputBundle) {
        final String name = outputBundle.readString();
        final int[] shape = outputBundle.readIntArray();
        final float[] data = outputBundle.readFloatArray();
        final NativeFloatTensor tensor = new NativeFloatTensor(shape);
        tensor.write(data, 0, data.length);
        return Pair.<String, FloatTensor>create(name, tensor);
    }

    private boolean readOutputUserBufferTensors(final JniOutputBundle outputBundle,
                                                final Map<String, ? extends UserBufferTensor> outputMap) {
        final int outputCount = outputBundle.readInt();
        for (int i = 0; i < outputCount; i++) {
            final String name = outputBundle.readString();
            if (!outputMap.containsKey(name)) {
                return false;
            }

            readUserBufferTensor(outputBundle, outputMap.get(name));
        }
        return true;
    }

    private void readUserBufferTensor(final JniOutputBundle outputBundle,
                                      final UserBufferTensor outputTensor) {
        if (outputTensor.getEncoding() == UserBufferTensor.Encoding.TF8) {
            if (!(outputTensor instanceof NativeTF8UserBufferTensor)) {
                throw SnpeError.getInstance().getIllegalError(SnpeError.Error.ILLEGAL_USER_BUFFER_TENSOR);
            }

            NativeTF8UserBufferTensor tf8Tensor = NativeTF8UserBufferTensor.class.cast(outputTensor);
            tf8Tensor.setStepExactly0(outputBundle.readInt());
            tf8Tensor.setQuantizedStepSize(outputBundle.readFloat());
        }
    }

    private List<ByteBuffer> writeInputTensors(final JniInputBundle bundle,
                                               final Map<String, NativeFloatTensor> tensorMap) {
        final int inputsCount = tensorMap.size();
        bundle.writeInt(inputsCount);
        final List<ByteBuffer> buffers = new LinkedList<>();
        for (String name : tensorMap.keySet()) {
            final ByteBuffer buffer = writeTensor(bundle, name, tensorMap.get(name));
            buffers.add(buffer);
        }
        return buffers;
    }

    private ByteBuffer writeTensor(final JniInputBundle bundle, final String name,
                                   final NativeFloatTensor tensor) {
        bundle.writeLong(tensor.getNativeHandle());
        bundle.writeString(name);

        final ByteBuffer buffer = tensor.asByteBuffer();
        bundle.writeInt(buffer.capacity());
        return buffer;
    }

    private void writeUserBufferTensors(final JniInputBundle bundle,
                                        final Map<String, ? extends UserBufferTensor> tensorMap) {
        final int inputsCount = tensorMap.size();
        bundle.writeInt(inputsCount);
        for (String name : tensorMap.keySet()) {
            bundle.writeLong(NativeUserBufferHandle.class.cast(tensorMap.get(name)).getNativeHandle());
            bundle.writeString(name);
        }
    }

    private void assertInputTensor(final String name, final Tensor tensor) {
        if (tensor == null) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_NULL, name);
        }

        if (!(tensor instanceof NativeFloatTensor)) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_INSTANCE, "" + NativeFloatTensor.class);
        }

        if (!mInputTensors.containsKey(name)) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_INPUT_NETWORK_TENSORS, name);
        }

        final int[] expectedShape = mInputTensors.get(name).getDims();
        TensorUtil.assertShape(name, expectedShape, tensor.getShape());
    }

    private void assertInputTensor(final String name, final UserBufferTensor tensor) {
        if (tensor == null) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_NULL, name);
        }

        if (!(tensor instanceof NativeUserBufferHandle)) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_INSTANCE, "" + NativeUserBufferHandle.class);
        }

        if (!mInputTensors.containsKey(name)) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_INPUT_NETWORK_TENSORS, name);
        }
    }

    private void assertOutputTensor(final String name, final UserBufferTensor tensor) {
        if (tensor == null) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_NULL, name);
        }

        if (!(tensor instanceof NativeUserBufferHandle)) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_OUTPUT_INSTANCE, "" + NativeUserBufferHandle.class);
        }

        if (!mOutputTensors.containsKey(name)) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_OUTPUT_NETWORK_TENSORS, name);
        }
    }
