package com.qualcomm.qti.snpe.internal;

import com.qualcomm.qti.snpe.SnpeError;

public final class TensorUtil {

    public static int sizeOf(final int[] shape) {
        int size = 1;
        for (int dimension : shape) {
            size *= dimension;
        }
        return size;
    }

    public static int position(int[] shape, int[] position) {
        final int[] lowerDimSizes = new int[shape.length];
        int size = 1;
        for (int i = shape.length - 1; i >= 0; i--) {
            lowerDimSizes[i] = size;
            size *= shape[i];
        }

        int positionIndex = 0;
        for (int i = 0; i < position.length; i++) {
            positionIndex += position[i] * lowerDimSizes[i];
        }
        return positionIndex;
    }

    public static void assertShape(final String tensor, final int[] expected, final int[] actual) {
        if (expected.length != actual.length) {
            throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_SHAPE, "Tensor " + tensor + " has unexpected shape! " +
                    "Expected: " + expected.length + ", Actual: " + actual.length);
        }

        for (int i = 0; i < expected.length; i++) {
            if (expected[i] != actual[i]) {
                throw SnpeError.getInstance().getInvalidInputError(SnpeError.Error.INVALID_TENSOR_DIMENSION, "Tensor " + tensor + " has unexpected " +
                        "dimension (" + i + ")! Expected: " + expected[i] + ", Actual: " + actual[i]);
            }
        }
    }

    private TensorUtil() {
        // Not meant to be instantiated.
    }
}
