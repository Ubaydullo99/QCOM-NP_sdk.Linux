package com.qualcomm.qti.snpe;

import java.util.Map;
import java.util.Set;
import java.nio.ByteBuffer;


/**
 * @addtogroup java_apis
 * @{
 */

/**
 * {@link NeuralNetwork} helper functions and properties.
 */
public interface NeuralNetwork {
    /**
     * Target runtime where neural network can be executed.
     *
     * @see com.qualcomm.qti.snpe.SNPE.NeuralNetworkBuilder#isRuntimeSupported(Runtime)
     */
    enum Runtime {
        CPU(0), /*** < CPU*/
        GPU(1), /*** < GPU with 16 bit data storage and 32 bit math*/
        DSP(2), /*** < DSP with both data storage and math in 8bit tensorflow style fixed point*/
        GPU_FLOAT16(3), /*** < GPU with 16 bit data storage and 16 bit math*/
        AIP(5); /*** < AIP*/

        public final int ordinal;

        /**
         * Select the runtime - CPU, GPU, DSP, GPU_FLOAT16, or AIP
         *
         * @param ordinal runtime value
         */
        Runtime(final int ordinal) {
            this.ordinal = ordinal;
        }
    }

    /**
     * Defines a performance profile for the runtime.
     */
    enum PerformanceProfile {
        DEFAULT(0),                     /**< DEFAULT */
        BALANCED(0),                    /**< BALANCED */
        HIGH_PERFORMANCE(1),            /**< HIGH_PERFORMANCE*/
        POWER_SAVER(2),                 /**< POWER_SAVER */
        SYSTEM_SETTINGS(3),             /**< SYSTEM_SETTINGS */
        SUSTAINED_HIGH_PERFORMANCE(4),  /**< SUSTAINED_HIGH_PERFORMANCE */
        BURST(5),                       /**< BURST */
        LOW_POWER_SAVER(6),             /**< LOW_POWER_SAVER */
        HIGH_POWER_SAVER(7),            /**< HIGH_POWER_SAVER */
        LOW_BALANCED(8);                /**< LOW_BALANCED */

        public final int ordinal;

        /**
         * Select the performance profile
         * <p>
         * DEFAULT - request for default performance mode (OS decision)
         * <p>
         * HIGH_PERFORMANCE - request for high performance mode
         *
         * @param ordinal performance profile
         */
        PerformanceProfile(final int ordinal) {
            this.ordinal = ordinal;
        }
    }

    /**
     * Execution priority hints
     */
    enum ExecutionPriorityHint {
        NORMAL(0), /*** <NORMAL*/
        HIGH(1), /*** <HIGH*/
        LOW(2), /*** <LOW*/
        NORMAL_HIGH(3); /***<NORMAL_HIGH*/

        public final int ordinal;

        /**
         * Select the execution priority hint
         * <p>
         * NORMAL - normal priority
         * HIGH - higher than normal priority
         * LOW - lower than normal priority
         * NORMAL_HIGH - between normal and high priority
         *
         * @param ordinal execution priority hint
         */
        ExecutionPriorityHint(final int ordinal) {
            this.ordinal = ordinal;
        }
    }

    enum RuntimeCheckOption {
        DEFAULT(0),
        NORMAL_CHECK(0),
        BASIC_CHECK(1),
        UNSIGNEDPD_CHECK(2);

        public final int ordinal;

        RuntimeCheckOption(final int ordinal){
            this.ordinal = ordinal;
        }
    }

    /**
     * Log level
     */
    enum LogLevel {
        LOG_FATAL(0),      /*** <LOG_FATAL*/
        LOG_ERROR(1),      /*** <LOG_ERROR*/
        LOG_WARN(2),       /*** <LOG_WARN*/
        LOG_VERBOSE(3),    /*** <LOG_VERBOSE*/
        LOG_INFO(4);       /*** <LOG_INFO*/

        public final int ordinal;

        /**
         * Select the logging level
         * <p>
         * LOG_FATAL - FATAL logging level
         * LOG_ERROR - ERROR logging level
         * LOG_WARN - WARN logging level
         * LOG_VERBOSE - VERBOSE logging level
         * LOG_INFO - INFO logging level
         *
         * @param ordinal log level
         */
        LogLevel(final int ordinal) {
            this.ordinal = ordinal;
        }
    }


    /**
     * @return The neural network's selected runtime.
     * @see com.qualcomm.qti.snpe.SNPE.NeuralNetworkBuilder#setRuntimeOrder(Runtime...)
     */
    NeuralNetwork.Runtime getRuntime();

    /**
     * @return The selected output layer(s) name(s).
     * @see com.qualcomm.qti.snpe.SNPE.NeuralNetworkBuilder#setOutputLayers(String...)
     */
    Set<String> getOutputLayers();

    /**
     * Forward propagates the inputs through the neural network.
     *
     * @param inputs map of input name and tensor.
     * @return a map of output name and tensor.
     */
    Map<String, FloatTensor> execute(Map<String, FloatTensor> inputs);

    /**
     * Forward propagates the inputs through the neural network.
     *
     * @param inputs map of input name and user buffer tensor.
     * @return a map of output name and user buffer tensor.
     */
    boolean execute(Map<String, ? extends UserBufferTensor> inputs,
                    Map<String, ? extends UserBufferTensor> outputs);

    /**
     * @return a map of the input tensors(s) shape(s).
     */
    Map<String, int[]> getInputTensorsShapes();

    /**
     * @return a map of the input tensors(s) shape(s).
     */
    Map<String, int[]> getOutputTensorsShapes();

    /**
     * @return a set with the input tensor(s) name(s).
     */
    Set<String> getInputTensorsNames();

    /**
     * @return a set with the output tensor(s) name(s).
     */
    Set<String> getOutputTensorsNames();

    /**
     * @param name name of the network tensor.
     * @return object representing the attributes of the tensor.
     */
    TensorAttributes getTensorAttributes(String name);

    /**
     * @return the user defined model version.
     */
    String getModelVersion();

    /**
     * Releases the neural network's resources. After this call you can no longer invoke any
     * methods on this class instance.
     */
    void release();

    /**
     * Creates a {@link FloatTensor} instance with the specified shape.
     *
     * @param shape tensor shape.
     * @return newly created instance.
     */
    FloatTensor createFloatTensor(int... shape);

    /**
     * Creates a {@link TF8UserBufferTensor} instance to wrap a user buffer.
     *
     * @param size              total size of the buffer, in bytes
     * @param strides           total number of bytes between elements in each dimension
     * @param stepExactly0      step value that represents exactly 0.0
     * @param quantizedStepSize step size
     * @param userBuffer        byte buffer that contains user data. This buffer must
     *                          have been allocated using direct allocation and possess
     *                          native byte ordering.
     * @return newly created instance
     */
    TF8UserBufferTensor createTF8UserBufferTensor(final int size, final int[] strides,
                                                  int stepExactly0, float quantizedStepSize,
                                                  ByteBuffer userBuffer);
}

 * @}
