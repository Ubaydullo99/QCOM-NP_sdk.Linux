package com.qualcomm.qti.snpe.internal.util;

import java.io.ByteArrayOutputStream;
import java.io.DataOutputStream;
import java.io.IOException;
import java.nio.ByteBuffer;
import java.nio.charset.Charset;
import java.util.Collection;

public final class JniInputBundle {

    private static final byte[] STRING_TERMINATOR = { 0 };
    private final ByteArrayOutputStream mBuffer;
    private final DataOutputStream mWriter;

    public JniInputBundle() {
        mBuffer = new ByteArrayOutputStream();
        mWriter = new DataOutputStream(mBuffer);
    }

    public void writeInt(final int value) {
        try {
            mWriter.writeInt(value);
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }

    public void writeInts(final int[] values) {
        writeInt(values.length);
        for (int value : values) {
            writeInt(value);
        }
    }

    public void writeLong(final long value) {
        try {
            mWriter.writeLong(value);
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }

    public void writeFloat(final float value) {
        try {
            mWriter.writeFloat(value);
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }

    public void writeStrings(final Collection<String> values) {
        writeInt(values.size());
        for (String string : values) {
            writeString(string);
        }
    }

    public void writeString(final String value) {
        final byte[] array = value.getBytes(Charset.forName("UTF-8"));
        writeInt(array.length + STRING_TERMINATOR.length);
        try {
            mWriter.write(array);
            mWriter.write(STRING_TERMINATOR);
        } catch (IOException e) {
            throw new IllegalStateException(e);
        }
    }

    public void writeBoolean(final boolean value) {
        writeInt(value ? 1 : 0);
    }

    public ByteBuffer marshall() {
        final ByteBuffer buffer = ByteBuffer.allocateDirect(mBuffer.size());
        buffer.put(mBuffer.toByteArray());
        buffer.flip();
        return buffer;
    }
}
