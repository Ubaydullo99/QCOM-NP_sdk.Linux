package com.qualcomm.qti.snpe.internal;

import com.qualcomm.qti.snpe.TF8UserBufferTensor;
import com.qualcomm.qti.snpe.SnpeError;
import com.qualcomm.qti.snpe.internal.util.JniInputBundle;

import java.nio.ByteBuffer;
import java.nio.ByteOrder;

final class NativeTF8UserBufferTensor extends TF8UserBufferTensor implements NativeUserBufferHandle {

    private static final int JNI_ERROR = -1;

    private long mNativeHandle;

    private final ByteBuffer mByteBuffer;

    private int mStepExactly0;

    private float mQuantizedStepSize;

    NativeTF8UserBufferTensor(final int size, final int[] strides,
                              int stepExactly0, float quantizedStepSize,
                              ByteBuffer byteBuffer) {
        super(size, strides);
        setStepExactly0(stepExactly0);
        setQuantizedStepSize(quantizedStepSize);

        mByteBuffer = byteBuffer;
        if (!mByteBuffer.isDirect()) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.BYTE_BUFFER_DIRECT_ERROR);
        }

        if (mByteBuffer.order() != ByteOrder.nativeOrder()) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.BYTE_BUFFER_ORDER_ERROR);
        }

        final JniInputBundle bundle = new JniInputBundle();
        bundle.writeInt(size);
        bundle.writeInts(strides);
        bundle.writeInt(stepExactly0);
        bundle.writeFloat(quantizedStepSize);

        final ByteBuffer buffer = bundle.marshall();
        try {
            mNativeHandle = nativeInit(buffer, buffer.capacity(), byteBuffer);
        } catch (IllegalStateException e) {
            throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
        }

        if (mNativeHandle == JNI_ERROR) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.TENSOR_CREATION_ERROR);
        }
    }

    public void setStepExactly0(int stepExactly0) {
        mStepExactly0 = stepExactly0;
    }

    public void setQuantizedStepSize(float quantizedStepSize) {
        mQuantizedStepSize = quantizedStepSize;
    }

    @Override
    public int getStepExactly0() {
        return mStepExactly0;
    }

    @Override
    public float getQuantizedStepSize() {
        return mQuantizedStepSize;
    }

    private native long nativeInit(ByteBuffer buffer, int capacity, ByteBuffer byteBuffer);

    @Override
    protected void finalize() throws Throwable {
        doRelease();
        super.finalize();
    }

    @Override
    public synchronized void release() {
        if (mNativeHandle == 0) {
            throw SnpeError.getInstance().getIllegalError(SnpeError.Error.USER_BUFFER_RELEASE_ERROR);
        } else {
            doRelease();
        }
    }

    private void doRelease() {
        if (mNativeHandle != 0) {
            try {
                nativeRelease(mNativeHandle);
            } catch (IllegalStateException e) {
                throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
            }

            mNativeHandle = 0;
        }
    }

    private native void nativeRelease(long nativeHandle);

    ByteBuffer asDirectByteBuffer() {
        return mByteBuffer;
    }

    @Override
    public long getNativeHandle() {
        return mNativeHandle;
    }
}
