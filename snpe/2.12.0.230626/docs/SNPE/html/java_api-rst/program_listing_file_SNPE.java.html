/**
 * {@link SNPE} class - top level APIs of the Snapdragon Neural Network Engine.
 */
public abstract class SNPE {

    /**
     * Log tag used internally by the SDK. @see {@link android.util.Log}
     */
    public static final String LOG_TAG = "snpe-android";

    private static String sRuntimeVersion;

    private static boolean sInitialized;

    private SNPE() { /*Not meant to be instantiated.*/ }

    private static synchronized void init(final Application application) {
        if (!sInitialized) {
            System.loadLibrary("snpe-android");
            try {
                SNPE.nativeInit(application.getApplicationInfo().nativeLibraryDir);
            } catch (IllegalStateException e){
                throw SnpeError.getInstance().getSnpeNativeError(e.getMessage());
            }

            sInitialized = true;
        }
    }

    /**
     * Returns the version of the Snapdragon Neural Network Engine.
     */
    public static String getRuntimeVersion(final Application application) {
        SNPE.init(application);
        if (sRuntimeVersion == null) {
            sRuntimeVersion = nativeGetRuntimeVersion();
        }
        return sRuntimeVersion;
    }

    /**
    * Registers a UDO Package Registration Library
    * @param application the application registering the package
    * @param regLibraryPath Path to registration library
    * @return true for successful registration, false for unsuccessful registration
    */
    public static boolean addOpPackage(final Application application, String regLibraryPath) {
       SNPE.init(application);
       if(regLibraryPath == null){
            return false;
       }
       return nativeAddOpPackage(regLibraryPath);
    }

    /**
     * Class exposes a way to configure and create {@link NeuralNetwork} instances.
     */
    public static class NeuralNetworkBuilder {

        public static class ModelNotSet extends RuntimeException {
        }

        private NeuralNetwork.Runtime[] mRuntimeOrder = new NeuralNetwork.Runtime[]{
                NeuralNetwork.Runtime.CPU
        };

        private PerformanceProfile mPerformanceProfile = DEFAULT;

        private ExecutionPriorityHint mExecutionPriorityHint = NORMAL;

        private final Map<String, int[]> mInputDimensions = new HashMap<>();

        private final Set<String> mOutputLayerNames = new HashSet<>();

        private boolean mIsDebugEnabled;

        private NativeBufferOutputStream mModelBuffer;

        private File mModelFile;

        private boolean mIsCpuFallbackEnabled;

        private boolean mUnsignedPD = false;

        private NeuralNetwork.RuntimeCheckOption mRuntimeCheckOption = NeuralNetwork.RuntimeCheckOption.NORMAL_CHECK;

        private boolean mUseUserSuppliedBuffers;

        private boolean mIsInitCacheEnabled;

        private boolean mCpuFixedPointMode = false;

        private boolean mIsDLCacheAvailable=true; // to avoid saving dl cache for runtimes other than DSP and AIP

        private String mStorageDirectory = "";

        private String mDlCachePath = "";

        public NeuralNetworkBuilder(final Application application) {
            SNPE.init(application);
        }

        /**
         * Sets the runtime selection order for the neural network.
         *
         * @param order array with the priority order for selecting an available runtime.
         * @return the builder itself.
         */
        public NeuralNetworkBuilder setRuntimeOrder(final NeuralNetwork.Runtime... order) {
            if (order == null || order.length == 0) {
                throw SnpeError.getInstance().getIllegalError(RUNTIME_ORDER_NULL_OR_EMPTY_ERROR);
            }

            for (NeuralNetwork.Runtime runtime : order) {
                if (runtime == null) {
                    throw SnpeError.getInstance().getIllegalError(RUNTIME_NULL_ERROR);
                }
            }

            mRuntimeOrder = new NeuralNetwork.Runtime[order.length];
            System.arraycopy(order, 0, mRuntimeOrder, 0, order.length);
            return this;
        }

        /**
         * Sets network's input dimensions to enable resizing of dimensions of the network.
         *
         * @param inputDimensions map of input names and their new dimensions.
         * @return the builder itself
         */
        public NeuralNetworkBuilder setInputDimensions(final Map<String, int[]> inputDimensions) {
            if (inputDimensions == null) {
                throw SnpeError.getInstance().getIllegalError(INPUT_DIMEN_NULL_OR_EMPTY_ERROR);
            }

            mInputDimensions.clear();
            mInputDimensions.putAll(inputDimensions);
            return this;
        }

        /**
         * Sets the output layers to return when {@link NeuralNetwork#execute(Map)} is called.
         *
         * @param outputLayers name of the output layers to retrieve.
         * @return the builder itself.
         */
        public NeuralNetworkBuilder setOutputLayers(final String... outputLayers) {
            if (outputLayers == null) {
                throw SnpeError.getInstance().getIllegalError(OUTPUT_LAYER_NULL_ERROR);
            }
            mOutputLayerNames.clear();
            mOutputLayerNames.addAll(Arrays.asList(outputLayers));
            return this;
        }

        /**
         * Reads the model from the provided {@link InputStream}. The caller is responsible
         * to close the stream after calling this method.
         * <b>After build() has been called, it is required that this method be invoked
         * to re-set the input stream before every successive call to build() on the same instance.</b>
         *
         * @param stream input stream from where to read the model from.
         * @param size   number of bytes to read from the stream.
         * @return the builder itself.
         * @throws IOException if not able to read from the stream.
         */
        public NeuralNetworkBuilder setModel(final InputStream stream, final int size)
                throws IOException {
            resetModel();
            mModelBuffer = IOUtil.bufferFromStream(stream, size);
            return this;
        }

        /**
         * Sets the file from where to read the DLC model from.
         *
         * @param file DLC model file.
         * @return the builder itself.
         * @throws IOException if not able to read from the stream.
         */
        public NeuralNetworkBuilder setModel(final File file) throws IOException {
            if (file == null) {
                throw SnpeError.getInstance().getIllegalError(MODEL_FILE_NULL_ERROR);
            }

            if (!file.exists()) {
                throw SnpeError.getInstance().getIOError(FILE_EXISTENCE_ERROR,file.getAbsolutePath());
            }

            if (!file.canRead()) {
                throw SnpeError.getInstance().getIOError(FILE_READ_ERROR,file.getAbsolutePath());
            }

            resetModel();
            mModelFile = file;
            return this;
        }

        public NeuralNetworkBuilder setDebugEnabled(final boolean isEnabled) {
            mIsDebugEnabled = isEnabled;
            return this;
        }

        /**
         * Selects a performance profile to use during the network execution.
         *
         * @param profile target performance profile.
         * @return the builder itself.
         */
        public NeuralNetworkBuilder setPerformanceProfile(final PerformanceProfile profile) {
            if (profile == null) {
                throw SnpeError.getInstance().getIllegalError(PERFORMANCE_PROFILE_NULL_ERROR);
            }
            mPerformanceProfile = profile;
            return this;
        }

        /**
         * Passes an execution priority hint to the runtime
         *
         * @param priority target execution priority.
         * @return the builder itself.
         */
        public NeuralNetworkBuilder setExecutionPriorityHint(final ExecutionPriorityHint priority) {
            if (priority == null) {
                throw SnpeError.getInstance().getIllegalError(EXECUTION_PRIORITY_NULL_ERROR);
            }
            mExecutionPriorityHint = priority;
            return this;
        }

        /**
         * Enables/Disables falling back to CPU if a layer is not supported in the
         * selected runtime.
         *
         * @param enabled
         */
        public NeuralNetworkBuilder setCpuFallbackEnabled(boolean enabled) {
            mIsCpuFallbackEnabled = enabled;
            return this;
        }

        /**
         * Sets Unsigned PD constraint while building network
         *
         * @param enabled
         */
        public NeuralNetworkBuilder setUnsignedPD(boolean enabled) {
            mUnsignedPD = enabled;
            return this;
        }

        /**
         * Sets a runtimeCheck option that is used for runtime availability check
         *
         * @param runtimeCheckOption runtime available check option
         */
        public NeuralNetworkBuilder setRuntimeCheckOption(final NeuralNetwork.RuntimeCheckOption runtimeCheckOption) {
            mRuntimeCheckOption = runtimeCheckOption;
            return this;
        }

        /**
         * Sets whether this network will perform inference with input from
         * user-supplied buffers.
         *
         * @param enabled
         */
        public NeuralNetworkBuilder setUseUserSuppliedBuffers(boolean enabled) {
            mUseUserSuppliedBuffers = enabled;
            return this;
        }


        /**
         * Enable/disable the functionality of init caching.
         * Init caching creates a set of init caches during network building/initialization process and adds them to DLC container. If such DLC container is saved by the user, in subsequent network building/initialization processes these init caches will be loaded from the DLC so as to reduce initialization time.
         *
         * @param isInitCacheEnabled enables/disables the functionality of init caching.
         * @param cacheTag The init caches are saved into a location under this tag name.
         * @param context The android context from which the network is created.
         *
         * @return The current instance of NeuralNetworkBuilder.
         */
        public NeuralNetworkBuilder setInitCacheEnabled(boolean isInitCacheEnabled, String cacheTag, Context context) throws SnpeError.IllegalArgsOrStateException {
            if (cacheTag == null) {
                throw SnpeError.getInstance().getIllegalError(CACHE_TAG_NULL_ERROR);
            } else if (cacheTag.trim().isEmpty()) {
                throw SnpeError.getInstance().getIllegalError(CACHE_TAG_EMPTY_ERROR);
            } else if (context == null) {
                throw SnpeError.getInstance().getIllegalError(CONTEXT_NULL_ERROR);
            }
            mDlCachePath = context.getFilesDir() + "/" + cacheTag + ".dlc";
            mIsInitCacheEnabled = isInitCacheEnabled;
            return this;
        }

        /**
         * Sets the fixed point execution of a quantized DLC on CPU runtime.
         * If a floating point DLC is executed with this option set, the
         * execution will be in floating point mode in CPU. If a quantized
         * DLC is executed without this option set, the execution will be
         * in floating point mode in CPU.
         *
         * @param cpuFxpMode Boolean If set to true, enables the fixed point mode.
         *
         * @return The current instance of NeuralNetworkBuilder.
         */
        public NeuralNetworkBuilder setCpuFixedPointMode(boolean fxpMode) {
            mCpuFixedPointMode = fxpMode;
            return this;
        }

        /**
         * Set the SNPE storage location for all SNPE instances in this process. This may only be called once, and if so must be called before creating any SNPE instances
         *
         * @param  Absolute path to a directory which SNPE may use for caching and other storage purposes.
         *
         * @return The current instance of NeuralNetworkBuilder.
         */
        public NeuralNetworkBuilder setStorageDirectory(String storageDirectory) throws SnpeError.SnpeIOException {

            if (storageDirectory == null) {
                throw SnpeError.getInstance().getIOError(DIRECTORY_PATH_NULL_ERROR);
            }

            if (storageDirectory.trim().isEmpty()) {
                throw SnpeError.getInstance().getIOError(DIRECTORY_PATH_EMPTY_ERROR);
            }

            File file = new File(storageDirectory);

            if (!file.exists()) {
                throw SnpeError.getInstance().getIOError(DIRECTORY_EXISTENCE_ERROR,file.getAbsolutePath());
            }

            if (!file.isDirectory()) {
                throw SnpeError.getInstance().getIOError(NOT_A_DIRECTORY_ERROR, file.getAbsolutePath());
            }

            if (!file.canRead()) {
                throw SnpeError.getInstance().getIOError(FILE_READ_ERROR, file.getAbsolutePath());
            }

            if (!file.canWrite()) {
                throw SnpeError.getInstance().getIOError(FILE_WRITE_ERROR,file.getAbsolutePath());
            }

            mStorageDirectory = storageDirectory;
            return this;
        }

        private void resetModel() {
            resetStreamModel();
            resetFileModel();
        }

        private void resetFileModel() {
            mModelFile = null;
        }

        private void resetStreamModel() {
            if (mModelBuffer != null) {
                mModelBuffer.release();
                mModelBuffer = null;
            }
        }

        /**
         * Builds a {@link NeuralNetwork} instance based on this builder's configuration.
         *
         * @return New network instance.
         */
        public NeuralNetwork build() {
            if((selectRuntime()== NeuralNetwork.Runtime.DSP||selectRuntime()== NeuralNetwork.Runtime.AIP) && mIsInitCacheEnabled){
                handleDLCaching();
            }

            if (mModelBuffer != null) {
                final NativeNetwork network = new NativeNetwork(selectRuntime(),
                        mModelBuffer, mOutputLayerNames, mIsDebugEnabled,
                        mPerformanceProfile, mExecutionPriorityHint,
                        mIsCpuFallbackEnabled, mUnsignedPD,
                        mUseUserSuppliedBuffers, mIsInitCacheEnabled,
                        mIsDLCacheAvailable, mCpuFixedPointMode,
                        mStorageDirectory, mDlCachePath, mInputDimensions);
                resetStreamModel();
                return network;
            } else if (mModelFile != null) {
                return new NativeNetwork(selectRuntime(), mModelFile,
                        mOutputLayerNames, mIsDebugEnabled, mPerformanceProfile,
                        mExecutionPriorityHint, mIsCpuFallbackEnabled,
                        mUnsignedPD, mUseUserSuppliedBuffers,
                        mIsInitCacheEnabled, mIsDLCacheAvailable,
                        mCpuFixedPointMode, mStorageDirectory,
                        mDlCachePath, mInputDimensions);
            } else {
                throw new ModelNotSet();
            }
        }

        private NeuralNetwork.Runtime selectRuntime() {
            NeuralNetwork.Runtime selected = null;
            for (NeuralNetwork.Runtime runtime : mRuntimeOrder) {
                if (isRuntimeSupported(runtime)) {
                    selected = runtime;
                    break;
                }
            }

            if (selected == null) {
                throw SnpeError.getInstance().getIllegalError(RUNTIME_TARGET_NOT_SUPPORTED_ERROR);
            }
            return selected;
        }

        public boolean isRuntimeSupported(final NeuralNetwork.Runtime runtime) {
            return NativeNetwork.isRuntimeAvailable(runtime, mRuntimeCheckOption);
        }


        /*Checks if a dlc cache file exists in app private directory or not.
         * If it doesn't exist, set initCacheEnabled to true
         * to save the cached file else load the cached file */
        private void handleDLCaching() {
                try {
                    File file = new File(mDlCachePath);
                    mIsDLCacheAvailable = file.exists();
                    if (mIsDLCacheAvailable) {
                        resetModel();
                        mModelFile = file;
                    }
                } catch (Exception ex) {
                    if (mIsDebugEnabled) {
                        Log.d(SNPE.LOG_TAG, ex.getMessage());
                    }
                }
        }
    }

    public static class Logger {
        private Logger() { }

        public boolean initializeLogging(final Application application, final NeuralNetwork.LogLevel logLevel) {
            SNPE.init(application);
            return nativeInitializeLogging(logLevel.ordinal());
        }

        public boolean setLogLevel(NeuralNetwork.LogLevel logLevel) {
            return nativeSetLogLevel(logLevel.ordinal());
        }

        public boolean terminateLogging() {
            return nativeTerminateLogging();
        }
    }

    public static Logger logger = new Logger();

    private static native String nativeGetRuntimeVersion();

    private static native void nativeInit(String nativeLibraryDir);

    private static native boolean nativeAddOpPackage(String regLibraryPath);

    private static native boolean nativeInitializeLogging(int logLevel);

    private static native boolean nativeSetLogLevel(int logLevel);

    private static native boolean nativeTerminateLogging();
}
