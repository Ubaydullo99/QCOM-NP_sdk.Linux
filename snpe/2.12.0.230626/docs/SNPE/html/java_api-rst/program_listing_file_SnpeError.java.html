    public static SnpeError getInstance() {
        if (mInstance == null) {
            mInstance = new SnpeError();

        return mInstance;

    public NativeException getSnpeNativeError(String message) {
        Error error = Error.getError(message);
        NativeException exception;
        if (error == null) {
            exception = new NativeException(0, message);
        } else {
            exception = new NativeException(error.code, "error_code : " + error.code + " error_message: " + error.message);
        }
        return exception;
    }

    public InvalidInputException getInvalidInputError(Error error, String info) {
        return new InvalidInputException(error.code, getErrorMessage(error.code, error.message, info));
    }

    public InvalidInputException getInvalidInputError(Error error) {
        return new InvalidInputException(error.code, getErrorMessage(error.code, error.message, ""));

        return "error_code : " + code + " error_message: " + message + info;

        // errors thrown from JNI layer
        private static Error getError(String message) {
            Error error = null;
            if (message.contains(NETWORK_BUNDLE_ACCESS_ERROR.message)) {
                error = NETWORK_BUNDLE_ACCESS_ERROR;
            } else if (message.contains(NETWORK_OBJECT_ACCESS_ERROR.message)) {
                error = NETWORK_OBJECT_ACCESS_ERROR;
            } else if (message.contains(NETWORK_RELEASE_ERROR.message)) {
                error = NETWORK_RELEASE_ERROR;
            } else if (message.contains(MODEL_READ_ERROR.message)) {
                error = MODEL_READ_ERROR;
            } else if (message.contains(BUFFER_STREAM_ACCESS_ERROR.message)) {
                error = BUFFER_STREAM_ACCESS_ERROR;
            } else if (message.contains(RUNTIME_ORDINAL_ERROR.message)) {
                error = RUNTIME_ORDINAL_ERROR;
            } else if (message.contains(BUFFER_ENCODING_ERROR.message)) {
                error = BUFFER_ENCODING_ERROR;
            } else if (message.contains(NETWORK_TENSOR_ACCESS_ERROR.message)) {
                error = NETWORK_TENSOR_ACCESS_ERROR;
            } else if (message.contains(INPUT_TENSOR_ACCESS_ERROR.message)) {
                error = INPUT_TENSOR_ACCESS_ERROR;
            } else if (message.contains(TENSOR_BUNDLE_ACCESS_ERROR.message)) {
                error = TENSOR_BUNDLE_ACCESS_ERROR;
            } else if (message.contains(TENSOR_CREATION_ERROR.message)) {
                error = TENSOR_CREATION_ERROR;
            } else if (message.contains(TENSOR_MANAGE_ERROR.message)) {
                error = TENSOR_MANAGE_ERROR;
            } else if (message.contains(BUFFER_BUNDLE_ACCESS_ERROR.message)) {
                error = BUFFER_BUNDLE_ACCESS_ERROR;
            } else if (message.contains(BUFFER_TENSOR_ADDRESS_ACCESS_ERROR.message)) {
                error = BUFFER_TENSOR_ADDRESS_ACCESS_ERROR;
            } else if (message.contains(BUFFER_OS_RETRIEVE_ERROR_MESSAGE.message)) {
                error = BUFFER_OS_RETRIEVE_ERROR_MESSAGE;
            } else if (message.contains(BUFFER_OS_SOURCE_ACCESS_ERROR_MESSAGE.message)) {
                error = BUFFER_OS_SOURCE_ACCESS_ERROR_MESSAGE;
            } else if (message.contains(BUFFER_OS_RELEASE_ERROR.message)) {
                error = BUFFER_OS_RELEASE_ERROR;
            } else if (message.contains(BUFFER_OS_SIZE_ERROR.message)) {
                error = BUFFER_OS_SIZE_ERROR;
            } else if (message.contains(JNI_INITIALIZE_ERROR.message)) {
                error = JNI_INITIALIZE_ERROR;
            } else if (message.contains(DSP_INITIALIZE_ERROR.message)) {
                error = DSP_INITIALIZE_ERROR;
            }

            return error;
        }

    class SnpeException extends RuntimeException {
        private int code;

        private SnpeException(int code, String detailMessage) {
            super(detailMessage);
            this.code = code;
        }

        public int getErrorCode() {
            return code;
        }
    }

    /**
     * Exception type for errors related to invalid inputs.
     * @details throws exception for invalid inputs with code and message.
     */
    public class InvalidInputException extends SnpeException {
        private InvalidInputException(int code, String detailMessage) {
            super(code, detailMessage);
        }
    }

    /**
     * Exception type for errors related to native network.
     * @details throws runtime exception with code and message for errors related to native network.
     */
    public class NativeException extends SnpeException {

        private NativeException(int code, String detailMessage) {
            super(code, detailMessage);
        }
    }

    /**
     * Exception type for errors related to IllegalState or IllegalArgument.
     *  @details throws runtime exception with code and message for errors related to illegal state or illegal arguments.
     */
    public class IllegalArgsOrStateException extends SnpeException {
        private IllegalArgsOrStateException(int code, String detailMessage) {
            super(code, detailMessage);
        }
    }

    /**
     * Exception type for errors related to IO Operations.
     * @details throws IO Exception with code and message
     */
    public class SnpeIOException extends SnpeException {
        private SnpeIOException(int code, String detailMessage) {
            super(code, detailMessage);
        }
    }
