package com.qualcomm.qti.snpe.internal;

import com.qualcomm.qti.snpe.TensorAttributes;
import com.qualcomm.qti.snpe.internal.util.JniOutputBundle;

import java.util.HashMap;
import java.util.Map;

final class TensorAttributesMap extends HashMap<String, TensorAttributes> {

    TensorAttributesMap(final JniOutputBundle bundle) {
        final int count = bundle.readInt();
        for (int i = 0; i < count; i++) {
            final String name = bundle.readString();
            final TensorAttributes attributes = new TensorAttributesImpl(bundle);
            put(name, attributes);
        }
    }

    public Map<String, int[]> getTensorShapes() {
        Map<String, int[]> tensorShapes = new HashMap<>();
        for (String name : keySet()) {
            tensorShapes.put(name, get(name).getDims());
        }
        return tensorShapes;
    }

}
