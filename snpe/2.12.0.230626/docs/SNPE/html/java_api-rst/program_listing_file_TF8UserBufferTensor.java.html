package com.qualcomm.qti.snpe;

/** @addtogroup java_apis
@{ */

/**
 * {@link UserBufferTensor} class representing TF8 user buffer elements.
 */
public abstract class TF8UserBufferTensor extends UserBufferTensor {

    public static final int TF8_MAX_VAL = 255;

    public static final int TF8_MIN_VAL = 0;

    protected TF8UserBufferTensor(final int size, final int[] strides) {
        super(size, strides);
    }

    /**
     * Returns the encoding type of the tensor.
     * @return the encoding type of the tensor
     */
    @Override
    public Encoding getEncoding(){
        return Encoding.TF8;
    }

    /**
     * Returns the step that represents exactly 0.0.
     * @return the step value
     */
    public abstract int getStepExactly0();

    /**
     * Returns the step size.
     * @return the step size
     */
    public abstract float getQuantizedStepSize();

    public float getMin() {
        return getQuantizedStepSize() * (0 - getStepExactly0());
    }

    public float getMax() {
        return getQuantizedStepSize() * (255 - getStepExactly0());
    }
}
