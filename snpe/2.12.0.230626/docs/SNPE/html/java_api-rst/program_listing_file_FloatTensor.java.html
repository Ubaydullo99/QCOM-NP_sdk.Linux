package com.qualcomm.qti.snpe;

/** @addtogroup java_apis
@{ */

/**
 * {@link Tensor} class containing float elements.
 */
public abstract class FloatTensor extends Tensor {

    protected FloatTensor(final int ...shape) {
        super(shape);
    }

    /**
     * Writes the range of the provided input array into the tensor at the specified position.
     * @param value float values to write.
     * @param offset start position to read from the input array.
     * @param length number of floats to read from the input array.
     * @param position the zero indexed position for each of the tensor dimensions to start writing
     *                 to.
     */
    public abstract void write(float[] value, int offset, int length, int... position);

    /**
     * Writes a single float value to the specified position.
     * @param value float value.
     * @param position the zero indexed position for each of the tensor dimensions to start writing
     *                 to.
     */
    public abstract void write(float value, int... position);

    /**
     * Reads elements from the tensor into the specified array range starting at the specified
     * tensor position.
     * @param value array to read values into.
     * @param offset start offset to write into array.
     * @param length number of elements to write into array.
     * @param position the zero indexed position for each of the tensor dimensions to start reading
     *                 from.
     * @return number of elements read.
     */
    public abstract int read(float[] value, int offset, int length, int... position);

    /**
     * Reads a single value from the tensor.
     *
     * @param position the zero indexed position for each of the tensor dimensions to start reading
     *                 from.
     * @return the value at the specified position.
     */
    public abstract float read(int... position);
}
