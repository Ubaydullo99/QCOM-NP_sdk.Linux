package com.qualcomm.qti.snpe;

import com.qualcomm.qti.snpe.internal.TensorUtil;

/** @addtogroup java_apis
@{ */

/**
 * {@link Tensor} class.
 */
public abstract class Tensor {

    private final int[] mShape;

    protected Tensor(final int ...shape) {
        mShape = new int[shape.length];
        System.arraycopy(shape, 0, mShape, 0, shape.length);
    }

    /**
     * Returns the shape of the tensor.
     * @return shape array.
     */
    public int[] getShape() {
        return mShape;
    }

    /**
     * Returns the number of elements in the tensor.
     * @return the size of the tensor.
     */
    public int getSize() {
        return TensorUtil.sizeOf(mShape);
    }

    /**
     * Releases the resources used by this tensor. After this call no other methods can be called
     * on this instance!
     */
    public abstract void release();
}

/** @} */ /* end addtogroup java_apis */
